var scope = "global scope";
function checkscope() {
    var scope = "local scope";
    function f() { return scope; }
    return f();
}
checkscope();

위 코드의 실행 결과는 "local scope" 이다. 이번엔 코드를 살짝 바꿔서

var scope = "global scope";
function checkscope() {
    var scope = "local scope";
    function f() { return scope; }
    return f;
}
checkscope()();

라고 했을 때 실행 결과는?

클로져 이해를 요구하는 괜찮은 예제인 듯 ㅋ.
