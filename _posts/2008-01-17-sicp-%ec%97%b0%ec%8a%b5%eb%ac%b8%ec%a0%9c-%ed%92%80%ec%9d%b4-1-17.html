SICP 연습문제 풀이 1.17

(define (double a) (* a 2))
(define (halve a) (/ a 2))
(define (mult a b)
  (cond ((= b 0) 0)
        ((even? b) (mult (double a) (halve b)))
        (else (+ a (mult a (- b 1))))))
