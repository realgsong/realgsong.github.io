1.22 의 풀이에 다음의 코드만 수정, 추가 한다.
[CODE]
(define (find-divisor n test-divisor)  
  (cond ((> (square test-divisor) n) n)  
        ((divides? test-divisor n) test-divisor)  
        (else (find-divisor n (next test-divisor)))))

(define (next n)
  (cond ((= n 2) 3)
        ((even? n) (+ n 1))
        (else (+ n 2))))
[/CODE]
